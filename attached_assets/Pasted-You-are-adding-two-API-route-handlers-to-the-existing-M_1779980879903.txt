-- 4A: IP HASHING UTILITY --

import crypto from 'crypto' (built-in, no install needed)

export function hashIp(rawIp) {
  if (!rawIp || typeof rawIp !== 'string') {
    return 'unknown'
  }
  // Strip port if present (e.g., "127.0.0.1:54321")
  const cleanIp = rawIp.split(':')[0]  
  return crypto
    .createHash('sha256')
    .update(cleanIp + process.env.IP_HASH_SALT || '')
    .digest('hex')
}
