--- STEP 2: THE CORRECT STACK FOR THIS DEPLOYMENT ---

Use this architecture:

  Database:     Supabase (PostgreSQL, hosted, free tier 
                sufficient for launch)
  ORM/Client:   @supabase/supabase-js (official client)
  Auth layer:   Supabase Row Level Security (RLS) — 
                configured via Supabase dashboard, 
                not in code
  Environment:  SUPABASE_URL and SUPABASE_ANON_KEY 
                stored as environment variables in 
                Vercel/Netlify dashboard AND in 
                local .env file

Why Supabase over alternatives:
  - Postgres, not a proprietary query language
  - Free tier: 500MB storage, 2GB bandwidth, 
    50,000 monthly active users — sufficient for 
    Morphix launch phase
  - Built-in dashboard to inspect rows without 
    writing admin routes
  - Works identically on Vercel, Netlify, and local dev
  - No connection pooling configuration needed at launch 
    (Supabase handles it)
  - If the project later moves to self-hosted infra, 
    the SQL schema is portable standard PostgreSQL

Do not use PlanetScale, Railway, or MongoDB.
Do not use Prisma at this stage — adds complexity 
without benefit for three tables.
Do not use the Supabase REST auto-API directly from 
the frontend — all DB access goes through server-side 
API routes only.

-- 4A: CLIENT INITIALIZATION --

import { createClient } from '@supabase/supabase-js'

Read SUPABASE_URL and SUPABASE_ANON_KEY from 
process.env.

Validate both on startup:
  if (!process.env.SUPABASE_URL || 
      !process.env.SUPABASE_ANON_KEY) {
    console.error(
      '[Morphix DB] FATAL: SUPABASE_URL and ' +
      'SUPABASE_ANON_KEY must be set in environment.'
    )
    process.exit(1)
  }

Initialize the client:
  const supabase = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_ANON_KEY
  )

Log on success:
  "[Morphix DB] Supabase client initialized."

-- 4B: EXPORTED QUERY HELPERS --

All helpers are async. All use try/catch AND check 
the Supabase error object (Supabase returns 
{ data, error } — always check error before data).

Helper 1: insertWaitlistSubscriber(data)
  Parameters: { full_name, company_name, 
                professional_email, inquiry_type }
  
  Operation:
    const { data, error } = await supabase
      .from('waitlist_subscribers')
      .insert([{
        full_name,
        company_name,
        professional_email,
        inquiry_type
      }])
      .select('id')
      .single()

  Error handling:
    if (error) {
      // Supabase returns code '23505' for unique violation
      if (error.code === '23505') {
        return { success: false, reason: 'EMAIL_EXISTS' }
      }
      console.error('[Morphix DB] insertWaitlistSubscriber:', 
                    error.message)
      return { success: false, reason: 'DB_ERROR', 
               detail: error.message }
    }
    return { success: true, id: data.id }

Helper 2: insertDataRoomRequest(data)
  Parameters: { investor_email, firm_name, 
                document_requested, access_token }
  
  Operation:
    const { data, error } = await supabase
      .from('investor_data_room_access')
      .insert([{
        investor_email,
        firm_name,
        document_requested,
        access_token
      }])
      .select('id')
      .single()

  Same error handling pattern as Helper 1.
  Unique violation on access_token returns:
    { success: false, reason: 'TOKEN_EXISTS' }

Helper 3: logCalculatorEvent(data)
  Parameters: { selected_hardware_tier,
                monthly_operational_hours,
                calculated_legacy_cost,
                calculated_omnidiff_cost,
                estimated_annual_savings,
                session_ip_hash }

  Operation:
    const { error } = await supabase
      .from('calculator_telemetry_logs')
      .insert([{ ...all fields }])

  This helper must never throw to the caller.
  On error: log silently, return { success: false }
  On success: return { success: true }

Helper 4: getWaitlistCount()
  Operation:
    const { count, error } = await supabase
      .from('waitlist_subscribers')
      .select('*', { count: 'exact', head: true })

  Returns: { count } on success
  Returns: { count: null, error: error.message } on failure

-- ================================================
-- Morphix Systems Inc. — Supabase Schema
-- Run this ONCE in Supabase SQL Editor
-- Never run this from application code
-- ================================================

CREATE TABLE IF NOT EXISTS waitlist_subscribers (
  id                  BIGSERIAL PRIMARY KEY,
  full_name           TEXT NOT NULL,
  company_name        TEXT,
  professional_email  TEXT NOT NULL UNIQUE,
  inquiry_type        TEXT NOT NULL DEFAULT 'General'
                      CHECK (inquiry_type IN 
                        ('General','Investor',
                         'Renter','Provider')),
  submission_timestamp TIMESTAMPTZ NOT NULL 
                        DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS investor_data_room_access (
  id                    BIGSERIAL PRIMARY KEY,
  investor_email        TEXT NOT NULL,
  firm_name             TEXT,
  document_requested    TEXT NOT NULL,
  access_token          TEXT NOT NULL UNIQUE,
  has_downloaded        BOOLEAN NOT NULL DEFAULT FALSE,
  last_active_timestamp TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS calculator_telemetry_logs (
  id                          BIGSERIAL PRIMARY KEY,
  selected_hardware_tier      TEXT NOT NULL,
  monthly_operational_hours   NUMERIC(10,2) NOT NULL,
  calculated_legacy_cost      NUMERIC(12,2) NOT NULL,
  calculated_omnidiff_cost    NUMERIC(12,2) NOT NULL,
  estimated_annual_savings    NUMERIC(12,2) NOT NULL,
  session_ip_hash             TEXT,
  calculation_timestamp       TIMESTAMPTZ NOT NULL 
                              DEFAULT NOW()
);

-- Row Level Security: enable but allow 
-- server-side service role full access.
-- Anon key gets INSERT only on waitlist 
-- and telemetry — no SELECT.
ALTER TABLE waitlist_subscribers 
  ENABLE ROW LEVEL SECURITY;
ALTER TABLE investor_data_room_access 
  ENABLE ROW LEVEL SECURITY;
ALTER TABLE calculator_telemetry_logs 
  ENABLE ROW LEVEL SECURITY;

-- Waitlist: allow anonymous inserts only
CREATE POLICY "anon_insert_waitlist"
  ON waitlist_subscribers FOR INSERT
  TO anon WITH CHECK (true);

-- Telemetry: allow anonymous inserts only
CREATE POLICY "anon_insert_telemetry"
  ON calculator_telemetry_logs FOR INSERT
  TO anon WITH CHECK (true);

-- Data room: no anon access at all
-- (handled server-side with service role key)
